    \section{Agile}

    \begin{frame}
        \begin{center}
            \includegraphics[width=\textwidth]{agileManifesto.png}
        \end{center}
        \begin{small}
            \url{http://agilemanifesto.org/iso/ru/manifesto.html}
        \end{small}
    \end{frame}

    \begin{frame}
        \frametitle{12 принципов Agile (1)}
        \begin{enumerate}
            \item Удовлетворение клиента за счёт ранней и бесперебойной поставки ценного программного обеспечения
            \item Приветствие изменений требований даже в конце разработки (это может повысить конкурентоспособность полученного продукта)
            \item Частая поставка рабочего программного обеспечения (каждый месяц или неделю или ещё чаще)
            \item Тесное, ежедневное общение заказчика с разработчиками на протяжении всего проекта
        \end{enumerate}
    \end{frame}

    \begin{frame}
        \frametitle{12 принципов Agile (2)}
        \begin{enumerate}
            \setcounter{enumi}{4}
            \item Проектом занимаются мотивированные личности, которые обеспечены нужными условиями работы, поддержкой и доверием
            \item Рекомендуемый метод передачи информации~--- личный разговор (лицом к лицу)
            \item Работающее программное обеспечение~--- лучший измеритель прогресса
            \item Инвесторы, разработчики и пользователи должны иметь возможность поддерживать постоянный темп на неопределённый срок
        \end{enumerate}
    \end{frame}

    \begin{frame}
        \frametitle{12 принципов Agile (3)}
        \begin{enumerate}
            \setcounter{enumi}{8}
            \item Постоянное внимание улучшению технического мастерства и удобному дизайну
            \item Простота --- искусство не делать лишней работы
            \item Лучшие технические требования, дизайн и архитектура получаются у самоорганизованной команды
            \item Постоянная адаптация к изменяющимся обстоятельствам. Команда должна систематически анализировать возможные способы улучшения эффективности и соответственно корректировать стиль своей работы
        \end{enumerate}
    \end{frame}

    \section{XP}

    \begin{frame}
        \frametitle{eXtreme Programming (XP)}
        \begin{outline}
            \1 Кент Бек, примерно 1999
            \1 Первая широкоизвестная agile-методология
            \1 Основные принципы: коммуникация, простота, обратная связь, храбрость
            \1 Непопулярна сейчас
                \2 Слишком революционные практики
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{Практики XP}
        \begin{enumerate}
            \item Короткий цикл обратной связи:
            \begin{outline}
                \1 Разработка через тестирование 
                \1 Игра в планирование
                \1 Заказчик всегда рядом
                \1 Парное программирование
            \end{outline}
        \end{enumerate}
    \end{frame}

    \begin{frame}
        \frametitle{1. Разработка через тестирование}
        \begin{columns}
            \begin{column}{0.7\textwidth}
                \begin{outline}
                    \1 Большое число автоматических тестов
                        \2 Модульные тесты
                        \2 Функциональные тесты
                    \1 100\% тестов должно проходить всегда
                    \1 Тесты как документирование кода
                    \1 Тесты как основа для рефакторинга
                    \1 Написание тестов перед написанием кода
                \end{outline}
            \end{column}
            \begin{column}{0.3\textwidth}
                \begin{center}
                    \includegraphics[width=\textwidth]{tddLoop.png}
                \end{center}
            \end{column}
        \end{columns}
    \end{frame}

    \begin{frame}
        \frametitle{2. Игра в планирование}
        \begin{outline}
            \1 Быстро получить приблизительный план работы
                \2 Последовательное уточнение
            \1 Идеальное время и load factor
            \1 Распределение ответственности между командой и заказчиком
                \2 Приоритизация задач
            \1 Customer stories
                \2 Estimatable
                \2 Testable
                \2 Bite-sizes
                \2 Progress
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{3. Заказчик всегда рядом}
        \begin{outline}
            \1 Представитель пользователей в команде
                \2 Принимает на себя ответственность
                \2 Влияет на ход разработки
            \1 Снижение затрат на коммуникацию
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{4. Парное программирование}
        \begin{outline}
            \1 Один компьютер, два программиста
                \2 Регулярное перемешивание пар
            \1 Более хороший дизайн и код
            \1 Повышение дисциплины
            \1 Коллективное владение кодом
            \1 Командный дух
            \1 Наставничество и обучение
            \1 Непрерывная социализация
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{Практики XP (2)}
        \begin{enumerate}
            \setcounter{enumi}{1}
            \item Непрерывный процесс:
            \begin{outline}
                \1 Непрерывная интеграция 
                \1 Рефакторинг 
                \1 Частые небольшие релизы
            \end{outline}
        \end{enumerate}
    \end{frame}

    \begin{frame}
        \frametitle{5. Непрерывная интеграция}
        \begin{outline}
            \1 Частые слияния веток разработки и сборки проекта
                \2 Несколько раз в день
                    \3 По внешнему запросу
                    \3 По расписанию
                    \3 По событию
            \1 Использование систем версионирования кода
            \1 Максимальная автоматизация
            \1 Постоянное наличие рабочей версии
            \1 Затраты на интеграцию
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{6. Рефакторинг}
        \begin{outline}
            \1 Код постоянно меняется, и это хорошо и правильно
            \1 Рефакторинг~--- средство поддержки эволюции
                \2 Долги проектирования
            \1 Реализуем только то, что нужно сейчас
                \2 Решаем проблемы по мере поступления
            \1 Нужно много хороших тестов
        \end{outline}
        \begin{center}
            \includegraphics[width=0.7\textwidth]{hatsRefactoring.png}
        \end{center}
    \end{frame}

    \begin{frame}
        \frametitle{7. Частые небольшие релизы}
        \begin{outline}
            \1 Меньше функциональности
                \2 Проще разрабатывать и тестировать
            \1 Быстрее доставка фич заказчику
            \1 Быстрее обратная связь от заказчика
            \1 Отказ от сдвига релизов
                \2 Урезаем функциональность
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{Практики XP (3)}
        \begin{enumerate}
            \setcounter{enumi}{2}
            \item Понимание, разделяемое всеми:
            \begin{outline}
                \1 Простота архитектуры
                \1 Метафора системы
                \1 Коллективное владение кодом
                \1 Стандарт кодирования
            \end{outline}
        \end{enumerate}
    \end{frame}

    \begin{frame}
        \frametitle{8. Простота архитектуры}
        \begin{outline}
            \1 Аллергия на BDUF\footnote{Big Design Up-Front}
            \1 Непрерывное проектирование
            \1 Самые простые решения
                \2 Ориентация на текущие задачи
                    \3 Простой не значит плохой
                    \3 Когда нельзя ничего выкинуть, чтобы не потерять функциональность
                \2 Непрерывный рефакторинг в зависимости от условий
            \1 Отказ от долгосрочного планирования
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{9. Метафора системы}
        \begin{outline}
            \1 Единое описание того, как работает система
                \2 Текстовое описание, понятное всем
            \1 Замена общепринятой архитектуры
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{10. Коллективное владение кодом}
        \begin{outline}
            \1 Разделение ответственности за весь код
                \2 Автор кода~--- вся команда
            \1 Каждый может менять любой код
                \2 Высокий bus factor 
                \2 Высокие требования к команде
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{11. Стандарт кодирования}
        \begin{outline}
            \1 Единый стандарт написания кода
                \2 Не тратим время на споры о несущественном
            \1 Простота изменений, интеграции, рефакторинга
            \1 Автоматизация проверок стиля
        \end{outline}
    \end{frame}

    \begin{frame}
        \frametitle{12. 40-часовая рабочая неделя}
        \begin{outline}
            \1 Авралы и переработки вредны на перспективе
                \2 Работоспособность падает
                \2 Выгорание
            \1 Жертвенность на работе~--- признак непрофессионализма
        \end{outline}
    \end{frame}
